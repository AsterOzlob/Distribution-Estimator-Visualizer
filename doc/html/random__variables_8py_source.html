48class SmoothedRandomVariable(RandomVariable, Estimation):
49 """
50 @brief Представляет сглаженную версию случайной величины с использованием оценки плотности ядра.
51
52 Этот класс расширяет RandomVariable и использует выборку для оценки pdf и cdf.
53 """
54
55 def __init__(self, sample: np.ndarray, h: float) -> None:
56 """
57 @brief Инициализирует SmoothedRandomVariable с выборкой и шириной полосы.
58
59 @param sample Массив выборочных точек данных.
60 @param h Ширина полосы, используемая в оценке плотности ядра.
61 """
62 super().__init__(sample)
63 self.h = h
64
65 def probability_density_function(self, x: float) -> float:
66 return np.mean([SmoothedRandomVariable._k((x - y) / self.h) for y in self.sample]) / self.h
67
68 def cumulative_distribution_function(self, x: float) -> float:
69 return np.mean([SmoothedRandomVariable._K((x - y) / self.h) for y in self.sample])
70
71 def quantile(self, alpha: float) -> None:
72 raise NotImplementedError
73
74 def _k(x: float) -> float:
75 if abs(x) <= 1:
76 return 0.75 * (1 - x * x)
77 else:
78 return 0
79
80 def _K(x: float) -> float:
81 if x < -1:
82 return 0
83 elif -1 <= x < 1:
84 return 0.5 + 0.75 * (x - x ** 3 / 3)

90 """
91 @brief Представляет нормальное (гауссово) распределение.
92
93 Характеризуется местоположением (средним) и масштабом (стандартным отклонением).
94 """
95
96 def __init__(self, location: float = 0, scale: float = 1) -> None:
97 """
98 @brief Инициализирует нормальную случайную величину с заданным местоположением и масштабом.
99
100 @param location Среднее нормального распределения.
101 @param scale Стандартное отклонение нормального распределения.
102 """
103 super().__init__()

105 self.scale = scale
106
107 def probability_density_function(self, x: float) -> float:
108 z = (x - self.location) / self.scale
109 return math.exp(-0.5 * z * z) / (math.sqrt(2 * math.pi) * self.scale)
110
111 def cumulative_distribution_function(self, x: float) -> float:
112 z = (x - self.location) / self.scale
113 if z <= 0:
114 return 0.852 * math.exp(-math.pow((-z + 1.5774) / 2.0637, 2.34))

117 def quantile(self, alpha: float) -> float:
118 return self.location + 4.91 * self.scale * (math.pow(alpha, 0.14) - math.pow(1 - alpha, 0.14))

129 """

135 super().__init__()

140 return 1.0 / (self.b - self.a) if self.a <= x <= self.b else 0
